"""Definitions for all core numeric instructions."""
import math

from pyshgp.push.instruction import SimpleInstruction
from pyshgp.utils import Token


def _add(a, b):
    return b + a,


def _sub(a, b):
    return b - a,


def _mult(a, b):
    return b * a,


def _p_div(a, b):
    if a == 0:
        return Token.revert
    return b / a,


def _p_mod(a, b):
    if a == 0:
        return Token.revert
    return b % a,


def _inc(x):
    return x + 1,


def _dec(x):
    return x - 1,
